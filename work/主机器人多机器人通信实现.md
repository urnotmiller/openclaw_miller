# 主机器人多机器人通信实现方案

## 概述

本方案将实现在主机器人（阿智）中与子机器人（代码专家、理财专家）的通信功能。我们将采用HTTP API通信方式，主机器人作为网关和协调者，负责：

1. 识别用户问题类型
2. 分发给对应的子机器人
3. 整合和呈现结果

#### 核心实现

```python
import requests
import json
import uuid
import time
import logging
from typing import Dict, Any, Optional

# 配置日志
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

# 子机器人配置
ROBOT_CONFIGS = {
    "code-expert": {
        "name": "代码专家",
        "url": "http://localhost:5001/api/chat",
        "description": "处理代码相关问题",
        "keywords": ["代码", "编程", "开发", "算法", "调试", "Python", "JavaScript", "Go"],
    },
    "finance-expert": {
        "name": "理财专家",
        "url": "http://localhost:5002/api/chat",
        "description": "处理理财和股票相关问题",
        "keywords": ["股票", "理财", "投资", "基金", "分析", "市场", "上证指数", "创业板"],
    }
}

# 请求和响应格式
class RequestMessage:
    def __init__(self, source: str, target: str, content: str, message_type: str = "question", context: Dict[str, Any] = None):
        self.id = str(uuid.uuid4())
        self.source = source
        self.target = target
        self.timestamp = int(time.time() * 1000)
        self.type = message_type
        self.content = content
        self.context = context or {}
    
    def to_dict(self) -> Dict[str, Any]:
        return {
            "id": self.id,
            "source": self.source,
            "target": self.target,
            "timestamp": self.timestamp,
            "type": self.type,
            "content": self.content,
            "context": self.context
        }

class ResponseMessage:
    def __init__(self, data: Dict[str, Any]):
        self.id = data.get("id")
        self.request_id = data.get("requestId")
        self.timestamp = data.get("timestamp")
        self.status = data.get("status")
        self.content = data.get("content")
        self.context = data.get("context", {})
    
    def is_success(self) -> bool:
        return self.status == "success"

def send_message(target: str, content: str, context: Dict[str, Any] = None) -> ResponseMessage:
    """
    发送消息给子机器人
    :param target: 目标机器人名称
    :param content: 消息内容
    :param context: 上下文信息
    :return: 响应消息
    """
    if target not in ROBOT_CONFIGS:
        logger.error(f"未知的目标机器人: {target}")
        return ResponseMessage({
            "id": str(uuid.uuid4()),
            "status": "error",
            "content": f"未知的目标机器人: {target}"
        })
    
    robot_config = ROBOT_CONFIGS[target]
    url = robot_config["url"]
    
    request_msg = RequestMessage(
        source="main",
        target=target,
        content=content,
        message_type="question",
        context=context
    )
    
    try:
        logger.info(f"发送请求到 {robot_config['name']}: {content[:100]}...")
        
        response = requests.post(
            url, 
            data=json.dumps(request_msg.to_dict()), 
            headers={"Content-Type": "application/json"},
            timeout=30
        )
        
        if response.status_code == 200:
            result_data = response.json()
            logger.info(f"收到来自 {robot_config['name']} 的响应: {result_data['content'][:100]}...")
            return ResponseMessage(result_data)
        else:
            logger.error(f"HTTP错误 {response.status_code} 从 {robot_config['name']}")
            return ResponseMessage({
                "id": request_msg.id,
                "status": "error",
                "content": f"HTTP错误: {response.status_code}"
            })
            
    except requests.exceptions.Timeout:
        logger.error(f"请求超时到 {robot_config['name']}")
        return ResponseMessage({
            "id": request_msg.id,
            "status": "error",
            "content": "请求超时"
        })
    except requests.exceptions.ConnectionError:
        logger.error(f"连接错误到 {robot_config['name']}")
        return ResponseMessage({
            "id": request_msg.id,
            "status": "error",
            "content": "连接失败"
        })
    except Exception as e:
        logger.error(f"请求失败到 {robot_config['name']}: {str(e)}")
        return ResponseMessage({
            "id": request_msg.id,
            "status": "error",
            "content": f"请求失败: {str(e)}"
        })

def recognize_question_type(question: str) -> Optional[str]:
    """
    识别问题类型，确定应该发给哪个子机器人
    :param question: 用户问题
    :return: 目标机器人名称，或者None（主机器人处理）
    """
    question_lower = question.lower()
    
    for robot, config in ROBOT_CONFIGS.items():
        keywords = config.get("keywords", [])
        for keyword in keywords:
            if keyword.lower() in question_lower:
                logger.info(f"识别到类型 '{robot}' 与关键词 '{keyword}'")
                return robot
    
    return None

def process_message(question: str, context: Dict[str, Any] = None) -> str:
    """
    处理用户消息
    :param question: 用户问题
    :param context: 上下文信息
    :return: 响应内容
    """
    logger.info(f"处理消息: {question[:100]}...")
    
    robot_name = recognize_question_type(question)
    
    if robot_name:
        logger.info(f"分发给子机器人: {robot_name}")
        response = send_message(robot_name, question, context)
        
        if response.is_success():
            return f"我已经让{ROBOT_CONFIGS[robot_name]['name']}来帮你回答这个问题:\n\n{response.content}"
        else:
            return f"很抱歉，我在处理请求时遇到了一些问题:\n\n{response.content}"
    else:
        logger.info("由主机器人处理")
        return handle_by_main_bot(question, context)

def handle_by_main_bot(question: str, context: Dict[str, Any] = None) -> str:
    """
    主机器人直接处理的问题
    :param question: 用户问题
    :param context: 上下文信息
    :return: 响应内容
    """
    greetings = ["你好", "您好", "hello", "hi"]
    
    if any(greeting in question.lower() for greeting in greetings):
        return "您好！我是阿智，你的全能助手。我可以帮你处理代码、理财和其他各类问题。有什么需要帮助的吗？"
    
    if "多机器人" in question.lower() or "子机器人" in question.lower():
        robot_info = "\n".join([
            f"• {config['name']}（处理{config['description']}）"
            for _, config in ROBOT_CONFIGS.items()
        ])
        
        return f"我可以协助你处理多种类型的问题，有以下专门的子机器人：\n\n{robot_info}\n\n你可以直接提问，我会帮你分发给对应的专家。"
    
    return "抱歉，我暂时无法处理这个问题。你可以尝试询问代码、理财相关的问题，或者简单的问候。"

def health_check() -> Dict[str, Any]:
    """
    健康检查
    :return: 各机器人健康状态
    """
    results = {
        "main": {
            "status": "healthy",
            "name": "主机器人",
            "message": "我是阿智，状态良好！"
        }
    }
    
    for robot, config in ROBOT_CONFIGS.items():
        try:
            response = send_message(robot, "健康检查")
            results[robot] = {
                "status": "healthy" if response.is_success() else "unhealthy",
                "name": config["name"],
                "message": response.content
            }
        except Exception as e:
            results[robot] = {
                "status": "unhealthy",
                "name": config["name"],
                "message": f"无法连接: {str(e)}"
            }
    
    return results

def generate_health_check_response() -> str:
    """
    生成健康检查响应字符串
    :return: 格式化的响应
    """
    results = health_check()
    lines = []
    
    for robot, info in results.items():
        status_icon = "✅" if info["status"] == "healthy" else "❌"
        lines.append(f"{status_icon} **{info['name']}**: {info['status']}")
        if info['message']:
            lines.append(f"   - {info['message']}")
    
    return "\n".join(lines)
```

## 4. 性能优化

### 4.1 并发处理

```python
from concurrent.futures import ThreadPoolExecutor
import multi_bot_communication as mbcomm

def process_messages_concurrently(messages):
    with ThreadPoolExecutor(max_workers=5) as executor:
        future_to_message = {
            executor.submit(mbcomm.process_message, message): message
            for message in messages
        }
        
        for future in future_to_message:
            try:
                result = future.result(timeout=30)
                print(f"处理完成: {result[:100]}")
            except Exception as e:
                print(f"处理失败: {e}")
```

### 4.2 缓存机制

```python
import redis
import json
import time

class ResponseCache:
    def __init__(self, host='localhost', port=6379, db=0):
        self.redis = redis.Redis(host=host, port=port, db=db)
    
    def get(self, key: str) -> Optional[str]:
        value = self.redis.get(key)
        if value:
            return value.decode('utf-8')
        return None
    
    def set(self, key: str, value: str, ttl: int = 600) -> None:
        self.redis.setex(key, ttl, value)
    
    def delete(self, key: str) -> None:
        self.redis.delete(key)

# 使用缓存的消息处理函数
cache = ResponseCache()

def process_cached_message(question: str) -> str:
    cache_key = f"msg_{hash(question)}"
    cached_response = cache.get(cache_key)
    
    if cached_response:
        logger.info(f"从缓存获取响应: {question[:100]}")
        return json.loads(cached_response)
    
    logger.info(f"重新处理问题: {question[:100]}")
    response = mbcomm.process_message(question)
    cache.set(cache_key, json.dumps(response), 600)
    
    return response
```
