# 子机器人直接通信指南

## 概述

你现在可以直接与子机器人进行通信，不需要经过主机器人（阿智）的转发。每个子机器人都有自己的API接口，可以直接通过HTTP请求进行通信。

## 通信方式

### 1. 发送消息
```bash
# 向代码专家发送消息
curl -X POST http://localhost:5001/api/chat \
  -H "Content-Type: application/json" \
  -d '{"content": "写一个阶乘函数"}'

# 向理财专家发送消息
curl -X POST http://localhost:5002/api/chat \
  -H "Content-Type: application/json" \
  -d '{"content": "分析股票市场"}'
```

### 2. 健康检查
```bash
# 检查代码专家健康状态
curl -X GET http://localhost:5001/health

# 检查理财专家健康状态
curl -X GET http://localhost:5002/health
```

## Python通信示例

### 基础通信
```python
import requests
import json

def send_message(robot_name, message):
    url = f"http://localhost:{5001 if robot_name == '代码专家' else 5002}/api/chat"
    
    payload = {
        "id": str(int(time.time())),
        "content": message,
        "timestamp": int(time.time()),
        "type": "question"
    }
    
    try:
        response = requests.post(url, json=payload, timeout=30)
        
        if response.status_code == 200:
            data = response.json()
            return data.get("content")
        else:
            return f"通信失败，状态码: {response.status_code}"
            
    except requests.exceptions.RequestException as e:
        return f"通信错误: {e}"

# 使用示例
if __name__ == "__main__":
    import time
    
    # 向代码专家发送消息
    code_response = send_message("代码专家", "写一个阶乘函数")
    print("代码专家:", code_response)
    
    # 向理财专家发送消息
    finance_response = send_message("理财专家", "分析股票市场")
    print("\n理财专家:", finance_response)
```

### 更高级的通信
```python
import requests
import time

class RobotClient:
    def __init__(self, name, port):
        self.name = name
        self.port = port
        
    def send_message(self, message):
        return send_message(self.name, message)
        
    def health_check(self):
        try:
            response = requests.get(f"http://localhost:{self.port}/health")
            return response.status_code == 200 and "healthy" in response.text
        except Exception:
            return False

class CodeExpertClient(RobotClient):
    def __init__(self):
        super().__init__("代码专家", 5001)
    
    def factorial(self, n):
        response = self.send_message(f"计算阶乘: {n}")
        return self._parse_code(response)
        
    def fibonacci(self, n):
        response = self.send_message(f"计算斐波那契数列: {n}")
        return self._parse_code(response)
        
    def hello(self):
        return self.send_message("你好")
        
    def _parse_code(self, response):
        # 简单的代码解析
        if "def" in response:
            return response.strip()
        return response

class FinanceExpertClient(RobotClient):
    def __init__(self):
        super().__init__("理财专家", 5002)
        
    def analyze_stock(self, symbol):
        return self.send_message(f"分析股票: {symbol}")
        
    def investment_advice(self, amount):
        return self.send_message(f"理财建议: {amount}")
        
    def hello(self):
        return self.send_message("你好")

if __name__ == "__main__":
    # 创建客户端
    code_expert = CodeExpertClient()
    finance_expert = FinanceExpertClient()
    
    print("=== 测试与子机器人通信 ===")
    
    # 测试代码专家
    print("\n1. 代码专家测试:")
    print(code_expert.hello())
    print("\n阶乘函数:")
    print(code_expert.factorial(5))
    
    # 测试理财专家
    print("\n2. 理财专家测试:")
    print(finance_expert.hello())
    print("\n股票分析:")
    print(finance_expert.analyze_stock("600000.SH"))
    
    # 健康检查
    print("\n3. 健康检查:")
    print(f"代码专家: {'✅ 健康' if code_expert.health_check() else '❌ 异常'}")
    print(f"理财专家: {'✅ 健康' if finance_expert.health_check() else '❌ 异常'}")
```

## 通信特点

### 代码专家通信特点
- 支持自然语言问题转化为代码
- 返回的代码具有良好的格式
- 对编程相关词汇敏感
- 可以回答关于Python、算法等问题

### 理财专家通信特点
- 回答理财相关问题
- 返回详细的分析和建议
- 包含风险提示信息
- 对股市、投资等词汇敏感

## 限制和注意事项

### 通信限制
1. 请求消息不应过长（建议不超过2000字符）
2. 响应时间通常不超过5秒
3. 支持HTTP POST和GET请求

### 最佳实践
- 使用try-except捕获通信异常
- 设置适当的超时时间（30秒左右）
- 验证响应的状态码
- 解析响应数据时进行适当的错误处理

## 通信示例脚本

### 使用Python脚本自动化通信
```python
#!/usr/bin/env python3
import requests
import time
from typing import Optional

class RobotCommunicator:
    def __init__(self, port: int):
        self.port = port
        
    def send_message(self, message: str) -> Optional[str]:
        url = f"http://localhost:{self.port}/api/chat"
        
        payload = {
            "id": str(int(time.time())),
            "content": message,
            "timestamp": int(time.time())
        }
        
        try:
            response = requests.post(
                url,
                json=payload,
                timeout=30
            )
            
            if response.status_code == 200:
                data = response.json()
                return data.get("content")
                
        except Exception as e:
            print(f"通信错误: {e}")
        
        return None

class DirectCommunication:
    def __init__(self):
        self.code_expert = RobotCommunicator(5001)
        self.finance_expert = RobotCommunicator(5002)
        
    def test_code_expert(self):
        print("=== 代码专家测试 ===")
        print(self.code_expert.send_message("你好"))
        print("\n阶乘函数:")
        print(self.code_expert.send_message("写一个阶乘函数"))
        
    def test_finance_expert(self):
        print("\n=== 理财专家测试 ===")
        print(self.finance_expert.send_message("你好"))
        print("\n股票分析:")
        print(self.finance_expert.send_message("分析股票市场"))
        
    def test_both(self):
        print("=== 同时测试两个子机器人 ===")
        
        from concurrent.futures import ThreadPoolExecutor
        
        with ThreadPoolExecutor() as executor:
            future1 = executor.submit(self.code_expert.send_message, "计算阶乘: 5")
            future2 = executor.submit(self.finance_expert.send_message, "分析股票: 600000.SH")
            
            print("代码专家:", future1.result())
            print("理财专家:", future2.result())

if __name__ == "__main__":
    comm = DirectCommunication()
    
    comm.test_code_expert()
    comm.test_finance_expert()
    comm.test_both()
```
